# Editable install of the single DSCraft package for local development.
# There is one installable package now (packages/dscraft); all nine
# subpackages (core, automl, clean, forecast, graph, vision, tune,
# security, agent) live inside it and are pulled in via the `all` extra.
#
# Usage: pip install -r requirements-dev.txt
#
# To install just one subpackage's extras instead of everything (e.g. to
# avoid pulling in torch for a forecast-only change), install the package
# directly with a narrower extras list, e.g.:
#   pip install -e "packages/dscraft[forecast,dev]"

-e packages/dscraft[all,dev]
